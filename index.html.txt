// 카메라 시작

const video =
document.getElementById("video");

const canvas =
document.getElementById("canvas");

const ctx =
canvas.getContext("2d");

function startCamera(){

    const camera = new Camera(video,{
        onFrame: async ()=>{
            await pose.send({image:video});
        },
        width:320,
        height:320
    });

    camera.start();
}

// =====================
// MediaPipe Pose
// =====================

const pose = new Pose({
    locateFile:(file)=>
    `https://cdn.jsdelivr.net/npm/@mediapipe/pose/${file}`
});

pose.setOptions({
    modelComplexity:1,
    smoothLandmarks:true,
    minDetectionConfidence:0.5,
    minTrackingConfidence:0.5
});

// =====================
// 실시간 균형 평가
// =====================

pose.onResults(results=>{

    canvas.width = video.videoWidth;
    canvas.height = video.videoHeight;

    ctx.drawImage(
        results.image,
        0,
        0,
        canvas.width,
        canvas.height
    );

    if(!results.poseLandmarks) return;

    let lm = results.poseLandmarks;

    let leftShoulder = lm[11];
    let rightShoulder = lm[12];

    let leftHip = lm[23];
    let rightHip = lm[24];

    // 어깨 균형
    let shoulderDiff =
    Math.abs(leftShoulder.y - rightShoulder.y);

    // 골반 균형
    let hipDiff =
    Math.abs(leftHip.y - rightHip.y);

    // 전체 균형 점수
    let totalBalance =
    shoulderDiff + hipDiff;

    let feedback = "";

    if(totalBalance < 0.05){

        feedback =
        "🟢 좋음 - 자세 균형 매우 안정적";

    }
    else if(totalBalance < 0.12){

        feedback =
        "🟡 보통 - 약간의 균형 차이";

    }
    else{

        feedback =
        "🔴 나쁨 - 자세 불균형 발생";

    }

    document.getElementById("feedback").innerHTML =
    feedback;

});
